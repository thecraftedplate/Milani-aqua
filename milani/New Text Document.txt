        const productCards = document.querySelectorAll('.product-card');
        const productDropdown = document.getElementById('productSelect');
        const notesTextarea = document.getElementById('notes');

        productCards.forEach(card => {
            card.addEventListener('click', function() {
                const size = this.getAttribute('data-size');
                const price = this.getAttribute('data-price');
                
                // 1. Set the dropdown choice
                productDropdown.value = size;
                
                // 2. Draft dynamic, clean premium notes configuration in USD
                notesTextarea.value = `I am selecting the premium ${size.toUpperCase()} bottle tier valued at ${price} per unit.\n\n[Please leave your specific requirements here, leaving space for additional request details]`;
                
                // 3. Smooth navigate into view
                document.getElementById('quote').scrollIntoView({ behavior: 'smooth' });
                
                // 4. Highlight field for the client
                setTimeout(() => {
                    document.getElementById('quantity').focus();
                }, 800);
            });
        });

        // Form submission setup
        document.getElementById('quoteForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const name = document.getElementById('fullName').value;
            const size = document.getElementById('productSelect').value;
            const qty = document.getElementById('quantity').value;
            
            alert(`Thank you, ${name}! Your custom allocation for ${qty} x ${size.toUpperCase()} bottles has been logged. Our logistics desk will contact you immediately.`);
            this.reset();
        });
